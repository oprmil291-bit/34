# services/mail_service.py
# Получение временной почты через API mail.tm (бесплатно)
# Arena отправит код подтверждения на эту почту

import aiohttp
import asyncio
import random
import string


MAIL_TM_API = "https://api.mail.tm"


async def get_random_password(length=14):
    """Генерирует случайный пароль"""
    chars = string.ascii_letters + string.digits + "!@#$%"
    return ''.join(random.choices(chars, k=length))


async def create_temp_email():
    """
    Создаёт временный email на mail.tm
    Возвращает: (email, password, token)
    """
    async with aiohttp.ClientSession() as session:
        # 1. Получаем доступный домен
        async with session.get(f"{MAIL_TM_API}/domains") as resp:
            domains = await resp.json()
            domain = domains["hydra:member"][0]["domain"]

        # 2. Генерируем случайный адрес
        username = f"arena_{random.randint(10000, 99999)}_{random.randint(100, 999)}"
        email = f"{username}@{domain}"
        password = await get_random_password()

        # 3. Регистрируем аккаунт на mail.tm
        async with session.post(
            f"{MAIL_TM_API}/accounts",
            json={"address": email, "password": password}
        ) as resp:
            if resp.status not in (200, 201):
                raise Exception(f"Не удалось создать почту: {resp.status}")

        # 4. Получаем токен для чтения писем
        async with session.post(
            f"{MAIL_TM_API}/token",
            json={"address": email, "password": password}
        ) as resp:
            token_data = await resp.json()
            token = token_data["token"]

        return email, password, token


async def wait_for_verification_code(mail_token: str, timeout=120):
    """
    Ждёт письмо с кодом подтверждения от Arena.
    Проверяет почту каждые 5 секунд.
    Возвращает код (строку из 6 цифр).
    """
    import re

    async with aiohttp.ClientSession() as session:
        headers = {"Authorization": f"Bearer {mail_token}"}

        for _ in range(timeout // 5):
            await asyncio.sleep(5)

            async with session.get(
                f"{MAIL_TM_API}/messages",
                headers=headers
            ) as resp:
                messages = await resp.json()

                if messages.get("hydra:totalItems", 0) > 0:
                    # Берём последнее письмо
                    msg_id = messages["hydra:member"][0]["id"]

                    async with session.get(
                        f"{MAIL_TM_API}/messages/{msg_id}",
                        headers=headers
                    ) as msg_resp:
                        msg_data = await msg_resp.json()
                        text = msg_data.get("text", "") + msg_data.get("html", "")

                        # Ищем 6-значный код
                        match = re.search(r'\b(\d{6})\b', text)
                        if match:
                            return match.group(1)

    return None  # Таймаут, код не пришёл

# services/captcha_solver.py
# Решение капчи через 2captcha.com
# Поддерживает Cloudflare Turnstile и hCaptcha

from twocaptcha import TwoCaptcha
import config
import asyncio


solver = TwoCaptcha(config.CAPTCHA_API_KEY)


async def solve_turnstile(site_url: str, site_key: str) -> str | None:
    """
    Решает Cloudflare Turnstile капчу.
    site_url — адрес страницы (например https://arena.ai/auth/signup)
    site_key — ключ капчи (находится в HTML коде страницы)
    Возвращает токен-решение или None если не удалось.
    """
    try:
        # 2captcha решает капчу асинхронно на своих серверах
        # Обычно занимает 10-30 секунд
        result = await asyncio.to_thread(
            solver.turnstile,
            sitekey=site_key,
            url=site_url
        )
        return result.get("code")
    except Exception as e:
        print(f"❌ Ошибка решения Turnstile: {e}")
        return None


async def solve_hcaptcha(site_url: str, site_key: str) -> str | None:
    """
    Решает hCaptcha (если Arena вдруг переключится на неё).
    """
    try:
        result = await asyncio.to_thread(
            solver.hcaptcha,
            sitekey=site_key,
            url=site_url
        )
        return result.get("code")
    except Exception as e:
        print(f"❌ Ошибка решения hCaptcha: {e}")
        return None

# Отправка сообщений в Arena.ai от лица зарегистрированного аккаунта

import aiohttp
import json


class ArenaClient:
    """Клиент для общения с Arena.ai через сохранённую сессию"""

    def __init__(self, session_data: dict, proxy: str, user_agent: str):
        self.cookies = {c["name"]: c["value"] for c in session_data.get("cookies", [])}
        self.storage = session_data.get("storage", {})
        self.proxy = proxy
        self.user_agent = user_agent

        # Пытаемся достать токен авторизации из localStorage
        self.auth_token = (
            self.storage.get("access_token")
            or self.storage.get("token")
            or self.storage.get("auth_token")
            or self.cookies.get("access_token", "")
        )

    async def send_message(self, prompt: str, model: str = "gpt-4o") -> str:
        """
        Отправляет сообщение в Arena и возвращает ответ.

        Возвращает:
        - Текст ответа (строка)
        - "SESSION_EXPIRED" если сессия протухла
        - "ERROR_XXX" при других ошибках
        """

        headers = {
            "User-Agent": self.user_agent,
            "Content-Type": "application/json",
            "Origin": "https://arena.ai",
            "Referer": "https://arena.ai/chat",
            "Accept": "application/json",
        }

        if self.auth_token:
            headers["Authorization"] = f"Bearer {self.auth_token}"

        # TODO: ПОДСМОТРЕТЬ НА САЙТЕ точную структуру JSON запроса
        # Это пример, реальный формат может отличаться
        payload = {
            "model": model,
            "messages": [
                {"role": "user", "content": prompt}
            ],
            "stream": False,
            "temperature": 0.7,
        }

        # TODO: ПОДСМОТРЕТЬ НА САЙТЕ точный URL API
        api_url = "https://arena.ai/api/v1/chat/completions"

        async with aiohttp.ClientSession(cookies=self.cookies) as session:
            try:
                async with session.post(
                    api_url,
                    json=payload,
                    headers=headers,
                    proxy=self.proxy if self.proxy else None,
                    timeout=aiohttp.ClientTimeout(total=60),
                ) as response:

                    if response.status == 200:
                        data = await response.json()
                        # TODO: ПОДСМОТРЕТЬ НА САЙТЕ структуру ответа
                        return data.get("choices", [{}])[0].get("message", {}).get("content", "Пустой ответ")

                    elif response.status in (401, 403):
                        return "SESSION_EXPIRED"

                    elif response.status == 429:
                        return "RATE_LIMITED"

                    else:
                        text = await response.text()
                        return f"ERROR_{response.status}: {text[:200]}"

            except aiohttp.ClientError as e:
                return f"CONNECTION_ERROR: {str(e)}"
            except Exception as e:
                return f"UNKNOWN_ERROR: {str(e)}"

    async def check_health(self) -> bool:
        """
        Быстрая проверка: жива ли сессия?
        Делает лёгкий запрос и проверяет, не вернулась ли 401.
        """
        headers = {
            "User-Agent": self.user_agent,
            "Authorization": f"Bearer {self.auth_token}",
        }

            try:
                # TODO: ПОДСМОТРЕТЬ НА САЙТЕ лёгкий эндпоинт (профиль, настройки)

                    "https://arena.ai/api/v1/user/profile",
                    headers=headers,
                    proxy=self.proxy if self.proxy else None,
                    timeout=aiohttp.ClientTimeout(total=10),
                ) as response:
                    return response.status == 200
            except:
                return False

# services/session_manager.py
# Управляет пулом аккаунтов: выдаёт, привязывает, заменяет протухшие

from sqlalchemy.orm import Session
from database import Account
from services.arena_client import ArenaClient
import random
import config


class SessionManager:
    """Менеджер сессий — мозг пула аккаунтов"""

    def __init__(self, db: Session):
        self.db = db

    def get_user_account(self, tg_id: int) -> Account | None:
        """Получить аккаунт, привязанный к пользователю"""
        return (
            self.db.query(Account)
            .filter(Account.assigned_to_tg == tg_id, Account.status == "active")
            .first()
        )

    def get_free_account(self) -> Account | None:
        """Получить случайный свободный аккаунт из пула"""
        free_accounts = (
            self.db.query(Account)
            .filter(Account.assigned_to_tg == None, Account.status == "active")
            .all()
        )
        if free_accounts:
            return random.choice(free_accounts)
        return None

    def assign_account_to_user(self, tg_id: int) -> Account | None:
        """
        Выдать пользователю аккаунт.
        Сначала ищет уже привязанный, потом свободный из пула.
        """
        # 1. Проверяем, есть ли уже привязанный
        account = self.get_user_account(tg_id)
        if account:
            return account

        # 2. Берём свободный из пула
        free = self.get_free_account()
        if free:
            free.assigned_to_tg = tg_id
            self.db.commit()
            self.db.refresh(free)
            return free

        # 3. Пул пуст — вернём None (бот запустит авторег)
        return None

    def mark_account_banned(self, account: Account):
        """Пометить аккаунт как забаненный и отвязать от пользователя"""
        account.status = "banned"
        account.assigned_to_tg = None
        self.db.commit()

    def mark_account_expired(self, account: Account):
        """Пометить аккаунт как протухший"""
        account.status = "expired"
        account.assigned_to_tg = None
        self.db.commit()

    def replace_account_for_user(self, tg_id: int) -> Account | None:
        """Заменить аккаунт пользователя на новый из пула"""
        old = self.get_user_account(tg_id)
        if old:
            self.mark_account_expired(old)
        return self.assign_account_to_user(tg_id)

    def get_pool_stats(self) -> dict:
        """Статистика пула (для админа)"""
        total = self.db.query(Account).count()
        active = self.db.query(Account).filter(Account.status == "active").count()
        free = self.db.query(Account).filter(
            Account.assigned_to_tg == None, Account.status == "active"
        ).count()
        banned = self.db.query(Account).filter(Account.status == "banned").count()
        return {
            "total": total,
            "active": active,
            "free": free,
            "banned": banned,
        }

    def get_arena_client(self, account: Account) -> ArenaClient:
        """Создать клиент Arena для конкретного аккаунта"""
        return ArenaClient(
            session_data=account.session_data,
            proxy=account.proxy,
            user_agent=account.user_agent,
        )

# utils/masking.py
# Маскировка ответов — чтобы пользователь не догадался, что внутри Arena.ai

import re


# Словарь замен: что убираем → на что заменяем
REPLACEMENTS = {
    # Названия платформ
    "LMSYS": "NeuralCore",
    "lmsys": "NeuralCore",
    "Arena.ai": "PrivateAI",
    "arena.ai": "PrivateAI",
    "Chatbot Arena": "AI Assistant",
    "LM Arena": "AI Engine",
    "lmarena": "AI Engine",

    # Названия моделей (чтобы не палить что это Arena)
    "Claude 3.5 Sonnet": "NeuroMax Pro",
    "Claude 3 Opus": "NeuroMax Ultra",
    "Claude": "NeuroMax",
    "GPT-4o": "Prime-X",
    "GPT-4": "Prime",
    "GPT-3.5": "Prime Lite",
    "Gemini 1.5 Pro": "Orion Pro",
    "Gemini": "Orion",
    "DeepSeek R1": "Insight Alpha",
    "DeepSeek": "Insight",
    "Llama 3": "OpenMind",
    "Llama": "OpenMind",
    "Mistral": "Zephyr",
    "Qwen": "Nexus",
}

# Ссылки и футеры, которые нужно удалить полностью
REMOVE_PATTERNS = [
    r"Join our Discord.*",
    r"Visit https?://\S*arena\S*",
    r"Powered by LMSYS.*",
    r"© \d{4} LMSYS.*",
    r"\[Learn more\]\(https?://\S+\)",
]


def mask_response(text: str) -> str:
    """
    Очищает ответ от любых упоминаний Arena, LMSYS и реальных названий моделей.
    Пользователь видит только твой "бренд".
    """
    if not text:
        return text

    result = text

    # 1. Заменяем названия
    for original, replacement in REPLACEMENTS.items():
        result = result.replace(original, replacement)

    # 2. Удаляем ссылки и футеры
    for pattern in REMOVE_PATTERNS:
        result = re.sub(pattern, "", result, flags=re.IGNORECASE | re.DOTALL)

    # 3. Убираем лишние пустые строки
    result = re.sub(r'\n{3,}', '\n\n', result)

    return result.strip()

@router.message(F.text)
async def handle_message(message: Message):
    """Главная функция — обработка всех текстовых сообщений"""
    db = SessionLocal()

    try:
        mgr = SessionManager(db)
        account = mgr.get_user_account(message.from_user.id)

        if not account:
            await message.answer("Сначала нажми /start для активации.")
            return

        # Показываем "печатает..."
        await bot.send_chat_action(message.chat.id, ChatAction.TYPING)

        # Отправляем запрос в Arena
        client = mgr.get_arena_client(account)
        response = await client.send_message(message.text)

        # Обработка ошибок
        if response == "SESSION_EXPIRED":
            logger.warning(f"Сессия протухла для аккаунта {account.email}")
            mgr.mark_account_expired(account)

            # Пытаемся выдать новый аккаунт
            new_acc = mgr.replace_account_for_user(message.from_user.id)
            if new_acc:
                client = mgr.get_arena_client(new_acc)
                response = await client.send_message(message.text)
            else:
                await message.answer("⏳ Обновляю подключение... Попробуй через минуту.")
                return

        elif response == "RATE_LIMITED":
            await message.answer("⏳ Слишком много запросов. Подожди 30 секунд.")
            return

        elif response.startswith("ERROR_"):
            logger.error(f"Ошибка Arena: {response}")
            await message.answer("⚠️ Временная ошибка. Попробуй ещё раз.")
            return

        # Маскируем ответ и отправляем пользователю
        clean_response = mask_response(response)

        # Telegram ограничивает сообщения 4096 символами
        if len(clean_response) > 4000:
            for i in range(0, len(clean_response), 4000):
                await message.answer(clean_response[i:i + 4000])
        else:
            await message.answer(clean_response)

        # Обновляем счётчик сообщений
        user = db.query(User).filter(User.telegram_id == message.from_user.id).first()
        if user:
            user.messages_count = (user.messages_count or 0) + 1
            db.commit()

    except Exception as e:
        logger.error(f"Ошибка обработки сообщения: {e}")
        await message.answer("⚠️ Что-то пошло не так. Попробуй ещё раз.")
    finally:
        db.close()

# worker.py
# Фоновый процесс, который следит за пулом аккаунтов.
# Если свободных аккаунтов мало — автоматически регистрирует новые.
# Запускается ОТДЕЛЬНО от бота: python worker.py

import asyncio
import logging
from database import init_db, SessionLocal, Account
from services.registrator import register_new_account
import config

logging.basicConfig(level=logging.INFO, format="%(asctime)s | WORKER | %(message)s")
logger = logging.getLogger(__name__)


async def maintain_pool():
    """Бесконечный цикл поддержания пула аккаунтов"""
    logger.info(f"🔄 Воркер запущен. Целевой размер пула: {config.MIN_POOL_SIZE}")

    while True:
        db = SessionLocal()
        try:
            # Считаем свободные аккаунты
            free_count = (
                db.query(Account)
                .filter(Account.assigned_to_tg == None, Account.status == "active")
                .count()
            )

            logger.info(f"📊 Свободных аккаунтов в пуле: {free_count}/{config.MIN_POOL_SIZE}")

            if free_count < config.MIN_POOL_SIZE:
                needed = config.MIN_POOL_SIZE - free_count
                logger.info(f"🔧 Нужно зарегистрировать {needed} новых аккаунтов")

                for i in range(needed):
                    # Берём прокси по кругу
                    if not config.PROXIES:
                        logger.error("❌ Нет прокси в конфиге!")
                        break

                    proxy = config.PROXIES[i % len(config.PROXIES)]
                    logger.info(f"🔧 Регистрация {i + 1}/{needed}...")

                    result = await register_new_account(proxy)

                    if result:
                        new_acc = Account(
                            email=result["email"],
                            password=result["password"],
                            session_data=result["session_data"],
                            user_agent=result["user_agent"],
                            proxy=result["proxy"],
                            status="active",
                            assigned_to_tg=None,  # Свободен!
                        )
                        db.add(new_acc)
                        db.commit()
                        logger.info(f"✅ Аккаунт {result['email']} добавлен в пул")
                    else:
                        logger.warning(f"⚠️ Не удалось зарегистрировать аккаунт")

                    # Пауза между регистрациями (чтобы не спалиться)
                    await asyncio.sleep(30)
            else:
                logger.info("✅ Пул в норме, отдыхаем 5 минут")

        except Exception as e:
            logger.error(f"❌ Ошибка воркера: {e}")
        finally:
            db.close()

        # Проверка каждые 5 минут
        await asyncio.sleep(300)


async def main():
    init_db()
    await maintain_pool()


if __name__ == "__main__":
    asyncio.run(main())
